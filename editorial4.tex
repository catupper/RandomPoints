\section{全点対の距離の平均}
\begin{align*}
  &E\left[\sum_{i=1}^N \sum_{j=i+1}^N | X_i - X_j | \right] \frac{2}{N(N-1)}\\
  &= \sum_{i=1}^N \sum_{j=i+1}^N E\left[|X_i - X_j|\right] \frac{2}{N(N-1)}
\end{align*}
ここで，$E\left[|X_i-X_j|\right]$は$i,j$によらないので
\begin{align*}
  &= E[|X_1 - X_2|] \\
  &= \frac13
\end{align*}
